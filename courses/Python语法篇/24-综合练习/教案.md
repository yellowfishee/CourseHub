### 一、基础复习题

**题目1：数字统计**

输入一行数字，统计正数、负数、零的个数。

```python
nums = list(map(int, input().split()))
positive = negative = zero = 0

for n in nums:
    if n > 0:
        positive += 1
    elif n < 0:
        negative += 1
    else:
        zero += 1

print(f"正数: {positive}, 负数: {negative}, 零: {zero}")
```

**题目2：字符串处理**

输入一个字符串，统计大小写字母、数字、其他字符的个数。

```python
s = input()
upper = lower = digit = other = 0

for ch in s:
    if ch.isupper():
        upper += 1
    elif ch.islower():
        lower += 1
    elif ch.isdigit():
        digit += 1
    else:
        other += 1

print(f"大写: {upper}, 小写: {lower}, 数字: {digit}, 其他: {other}")
```

**题目3：列表操作**

输入n个数字，找出第二大的数。

```python
n = int(input())
nums = list(map(int, input().split()))

# 方法1：排序
nums = list(set(nums))  # 去重
nums.sort()
print(nums[-2])

# 方法2：一次遍历
first = second = float('-inf')
for num in nums:
    if num > first:
        second = first
        first = num
    elif num > second and num != first:
        second = num
print(second)
```

### 二、挑战题

**题目1：回文判断**

判断一个字符串是否是回文（正读反读相同）。

```python
def is_palindrome(s):
    # 方法1：切片反转
    return s == s[::-1]

# 方法2：双指针
def is_palindrome(s):
    left, right = 0, len(s) - 1
    while left < right:
        if s[left] != s[right]:
            return False
        left += 1
        right -= 1
    return True

print(is_palindrome("abcba"))  # True
print(is_palindrome("hello"))  # False
```

**题目2：素数判断**

判断一个数是否是素数。

```python
import math

def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(math.sqrt(n)) + 1):
        if n % i == 0:
            return False
    return True

# 输出100以内的素数
primes = [i for i in range(2, 100) if is_prime(i)]
print(primes)
```

**题目3：冒泡排序**

实现冒泡排序算法。

```python
def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
    return arr

nums = [64, 34, 25, 12, 22, 11, 90]
print(bubble_sort(nums))
```

### 三、竞赛真题选讲

**题目1：水仙花数**

找出所有三位水仙花数（各位数字的立方和等于本身）。

```python
for n in range(100, 1000):
    a = n // 100        # 百位
    b = (n // 10) % 10  # 十位
    c = n % 10          # 个位
    
    if a**3 + b**3 + c**3 == n:
        print(n)
# 输出: 153, 370, 371, 407
```

**题目2：完数**

找出1000以内的完数（等于其所有真因子之和）。

```python
def is_perfect(n):
    if n < 2:
        return False
    total = 1  # 1是所有大于1的数的因子
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            total += i
            if i != n // i:
                total += n // i
    return total == n

for n in range(1, 1000):
    if is_perfect(n):
        print(n)
# 输出: 6, 28, 496
```

**题目3：最大公约数与最小公倍数**

```python
def gcd(a, b):
    """最大公约数 - 辗转相除法"""
    while b:
        a, b = b, a % b
    return a

def lcm(a, b):
    """最小公倍数"""
    return a * b // gcd(a, b)

a, b = 12, 18
print(f"最大公约数: {gcd(a, b)}")  # 6
print(f"最小公倍数: {lcm(a, b)}")  # 36
```

### 四、综合应用题

**题目1：学生成绩管理**

```python
students = {}

def add_student(name, scores):
    students[name] = scores

def get_average(name):
    if name in students:
        return sum(students[name]) / len(students[name])
    return None

def get_highest():
    best_student = None
    best_avg = 0
    for name, scores in students.items():
        avg = sum(scores) / len(scores)
        if avg > best_avg:
            best_avg = avg
            best_student = name
    return best_student, best_avg

# 测试
add_student("小明", [90, 85, 88])
add_student("小红", [95, 92, 90])
add_student("小刚", [80, 85, 82])

print(get_average("小明"))  # 87.666...
print(get_highest())  # ('小红', 92.333...)
```

**题目2：简单的通讯录**

```python
contacts = {}

def add_contact(name, phone):
    contacts[name] = phone

def find_contact(name):
    return contacts.get(name, "未找到")

def list_contacts():
    for name, phone in contacts.items():
        print(f"{name}: {phone}")

# 测试
add_contact("小明", "12345678")
add_contact("小红", "87654321")
print(find_contact("小明"))  # 12345678
list_contacts()
```
