🔄 三位数反转

n = int(input())
# 方法：字符串反转
result = int(str(n)[::-1])

🐔 鸡兔同笼

n, m = map(int, input().split())
y = (m - 2*n) // 2 # 兔子
x = n - y # 鸡

if y >= 0 and x >= 0:
  print(x, y)
else:
  print("No answer")
