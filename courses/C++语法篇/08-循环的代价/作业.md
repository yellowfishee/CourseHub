// 题目1：空循环计时
#include <stdio.h>
#include <time.h>
int main() {
    clock_t start = clock();
    for(int i = 0; i < 1000000; i++) {
        // 空循环
    }
    clock_t end = clock();
    printf("Time used = %.6f\n", (double)(end - start) / CLOCKS_PER_SEC);
    return 0;
}

// 题目2：阶乘之和（每步取模）
#include <stdio.h>
int main() {
    const int MOD = 1000000;
    int n, S = 0;
    scanf("%d", &n);
    
    for(int i = 1; i <= n; i++) {
        int factorial = 1;
        for(int j = 1; j <= i; j++) {
            factorial = (factorial * j) % MOD;
        }
        S = (S + factorial) % MOD;
    }
    
    printf("%d\n", S);
    return 0;
}

// 题目3：优化版本
#include <stdio.h>
int main() {
    const int MOD = 1000000;
    int n, S = 0;
    scanf("%d", &n);
    
    if(n > 25) n = 25;  // 优化！
    
    for(int i = 1; i <= n; i++) {
        int factorial = 1;
        for(int j = 1; j <= i; j++) {
            factorial = (factorial * j) % MOD;
        }
        S = (S + factorial) % MOD;
    }
    
    printf("%d\n", S);
    return 0;
}

// 题目4：时间比较
#include <stdio.h>
#include <time.h>
int main() {
    const int MOD = 1000000;
    
    // 算法A：计算到10000
    clock_t start = clock();
    int S = 0;
    for(int i = 1; i <= 10000; i++) {
        int factorial = 1;
        for(int j = 1; j <= i; j++) {
            factorial = (factorial * j) % MOD;
        }
        S = (S + factorial) % MOD;
    }
    clock_t end = clock();
    printf("算法A (10000): %.6f秒, 结果=%d\n", 
           (double)(end - start) / CLOCKS_PER_SEC, S);
    
    // 算法B：只计算到25
    start = clock();
    S = 0;
    for(int i = 1; i <= 25; i++) {
        int factorial = 1;
        for(int j = 1; j <= i; j++) {
            factorial = (factorial * j) % MOD;
        }
        S = (S + factorial) % MOD;
    }
    end = clock();
    printf("算法B (25): %.6f秒, 结果=%d\n", 
           (double)(end - start) / CLOCKS_PER_SEC, S);
    
    return 0;
}
