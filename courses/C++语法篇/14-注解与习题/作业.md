```c
// 题目1：最小元素与最后一个交换
#include <stdio.h>
int main() {
    int n, a[100];
    scanf("%d", &n);
    for(int i = 0; i < n; i++) scanf("%d", &a[i]);
    int minIdx = 0;
    for(int i = 1; i < n; i++)
        if(a[i] < a[minIdx]) minIdx = i;
    int temp = a[minIdx];
    a[minIdx] = a[n-1];
    a[n-1] = temp;
    for(int i = 0; i < n; i++) printf("%d ", a[i]);
    printf("\n");
    return 0;
}

// 题目2：判断是否全为数字
#include <stdio.h>
#include <ctype.h>
int main() {
    char s[100];
    scanf("%s", s);
    int ok = 1;
    for(int i = 0; s[i]; i++)
        if(!isdigit(s[i])) { ok = 0; break; }
    printf(ok ? "YES\n" : "NO\n");
    return 0;
}

// 题目3：统计元音字母
#include <stdio.h>
#include <ctype.h>
int main() {
    char s[100];
    fgets(s, sizeof(s), stdin);
    int cnt = 0;
    for(int i = 0; s[i]; i++) {
        char c = tolower(s[i]);
        if(c=='a'||c=='e'||c=='i'||c=='o'||c=='u') cnt++;
    }
    printf("%d\n", cnt);
    return 0;
}

// 习题3-6 回文串
#include <stdio.h>
#include <string.h>
int main() {
    char s[100];
    scanf("%s", s);
    int len = strlen(s);
    int ok = 1;
    for(int i = 0; i < len/2; i++)
        if(s[i] != s[len-1-i]) { ok = 0; break; }
    printf(ok ? "YES\n" : "NO\n");
    return 0;
}

// 习题3-7 单词计数
#include <stdio.h>
int main() {
    char s[100];
    int cnt = 0;
    while(scanf("%s", s) == 1) cnt++;
    printf("%d\n", cnt);
    return 0;
}
```
