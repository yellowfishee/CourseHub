#include <stdio.h>
int main() {
    int a, b, c, t;
    scanf("%d%d%d", &a, &b, &c);
    if (a > b) { t = a; a = b; b = t; }
    if (a > c) { t = a; a = c; c = t; }
    if (b > c) { t = b; b = c; c = t; }
    printf("%d %d %d\n", a, b, c);
    return 0;
}

// 习题1-6 三角形
#include <stdio.h>
int main() {
    int a, b, c;
    scanf("%d%d%d", &a, &b, &c);
    // 判断能否构成三角形
    if (a + b <= c || a + c <= b || b + c <= a) {
        printf("not a triangle\n");
    } else {
        // 判断是否为直角三角形
        int max = a;
        if (b > max) max = b;
        if (c > max) max = c;
        int x, y;
        if (max == a) { x = b; y = c; }
        else if (max == b) { x = a; y = c; }
        else { x = a; y = b; }
        if (max * max == x * x + y * y) {
            printf("yes\n");
        } else {
            printf("no\n");
        }
    }
    return 0;
}

    }
    return 0;
}
