// 题目1：三种循环
#include <stdio.h>
int main() {
    int sum = 0, i;
    
    // for循环
    for(i = 1; i <= 100; i++) sum += i;
    printf("for: %d\n", sum);
    
    // while循环
    sum = 0; i = 1;
    while(i <= 100) { sum += i; i++; }
    printf("while: %d\n", sum);
    
    // do-while循环
    sum = 0; i = 1;
    do { sum += i; i++; } while(i <= 100);
    printf("do-while: %d\n", sum);
    
    return 0;
}

        int found = 0;
        for(int i = 10; i <= 100; i++) {
            if(i % 3 == a && i % 5 == b && i % 7 == c) {
                printf("Case %d: %d\n", ++kase, i);

}

    return 0;
}

    }
