#include <stdio.h>
int main() {
    int n, x, kase = 0;
    while(scanf("%d", &n) == 1 && n) {
        int sum = 0, max = -1000000000;
        for(int i = 0; i < n; i++) {
            scanf("%d", &x);
            sum += x;
            if(x > max) max = x;

        printf("Case %d: max=%d avg=%.3f\n", ++kase, max, (double)sum/n);

// 题目4（正确代码）
#include <stdio.h>
#define INF 1000000000
int main() {
    int x, min = INF, max = -INF, sum = 0, count = 0;
    while(scanf("%d", &x) == 1) {
        if(x < min) min = x;
        if(x > max) max = x;
        sum += x;
        count++;
    }
    if(count > 0)
        printf("min=%d max=%d avg=%.3f\n", min, max, (double)sum/count);
    return 0;
}

#include <stdio.h>
int main() {
    int a, b, c, kase = 0;
    while(scanf("%d%d%d", &a, &b, &c) == 3) {
        int found = 0;
        for(int i = 10; i <= 100; i++) {
            if(i % 3 == a && i % 5 == b && i % 7 == c) {
                printf("Case %d: %d\n", ++kase, i);
                found = 1;
                break;
            }
        }
        if(!found) {
            printf("Case %d: No answer\n", ++kase);
        }
    }
    return 0;
}

    int n;
    scanf("%d", &n);
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < i; j++) printf(" ");
        for(int j = 0; j < 2*(n-i)-1; j++) printf("#");
        printf("\n");
    }
    return 0;
}

// 习题2-4 子序列的和
#include <stdio.h>
int main() {
    int n, m, kase = 0;
    while(scanf("%d%d", &n, &m) == 2 && n && m) {
        double sum = 0;
        for(int i = n; i <= m; i++) {
            sum += 1.0 / (i * i);
        }
        if(kase) printf("\n");
        printf("Case %d: %.5f\n", ++kase, sum);
    }
    return 0;
}
