## 参考答案

**选择题：** 1-B, 2-C, 3-B, 4-B, 5-B

**填空题：**
6. %d
7. %lf 或 %f
8. n*(n+1)/2
9. 格式
10. 1.600

**编程题参考答案：**

```c
// 题目1：平均数
#include <stdio.h>
int main() {
    double a, b, c;
    scanf("%lf %lf %lf", &a, &b, &c);
    printf("%.3f\n", (a + b + c) / 3);
    return 0;
}

// 题目2：温度转换
#include <stdio.h>
int main() {
    double f, c;
    scanf("%lf", &f);
    c = 5.0 * (f - 32.0) / 9.0;
    printf("%.3f\n", c);
    return 0;
}

// 题目3：连续和
#include <stdio.h>
int main() {
    int n;
    scanf("%d", &n);
    printf("%d\n", n * (n + 1) / 2);
    return 0;
}

// 题目4：打折
#include <stdio.h>
int main() {
    int n;
    double total;
    scanf("%d", &n);
    total = n * 95.0;
    if (total >= 300.0) {
        total *= 0.85;
    }
    printf("%.2f\n", total);
    return 0;
}

// 题目5：交换变量
#include <stdio.h>
int main() {
    int a, b, t;
    scanf("%d %d", &a, &b);
    t = a;
    a = b;
    b = t;
    printf("%d %d\n", a, b);
    return 0;
}
```
