**编程题（参考答案）：**

```c
// 题目1：统计字符串字符个数
#include <stdio.h>
#include <string.h>
int main() {
    char s[100];
    scanf("%s", s);
    printf("%lu\n", strlen(s));
    return 0;
}

// 题目2：比较两个字符串
#include <stdio.h>
#include <string.h>
int main() {
    char s1[100], s2[100];
    scanf("%s%s", s1, s2);
    if(strcmp(s1, s2) == 0)
        printf("YES\n");
    else
        printf("NO\n");
    return 0;
}

// 题目3：复制字符串
#include <stdio.h>
#include <string.h>
int main() {
    char s1[100], s2[100];
    scanf("%s", s1);
    strcpy(s2, s1);
    printf("%s\n", s2);
    return 0;
}

// 题目4：连接字符串
#include <stdio.h>
#include <string.h>
int main() {
    char s1[100], s2[100];
    scanf("%s%s", s1, s2);
    strcat(s1, s2);
    printf("%s\n", s1);
    return 0;
}

// 习题3-4 单词替换
#include <stdio.h>
#include <string.h>
int main() {
    char s[1000];
    fgets(s, sizeof(s), stdin);
    for(int i = 0; s[i]; i++) {
        if(i == 0 && s[i] == 'a' && s[i+1] == 'd' && s[i+2] == 'a')
            printf("Ada"), i += 2;
        else if(i > 0 && s[i-1] == ' ' && s[i] == 'a' && s[i+1] == 'd' && s[i+2] == 'a')
            printf("Ada"), i += 2;
        else
            printf("%c", s[i]);
    }
    return 0;
}

// 习题3-5 周期串
#include <stdio.h>
#include <string.h>
int main() {
    char s[100];
    scanf("%s", s);
    int n = strlen(s);
    for(int k = 1; k <= n; k++) {
        if(n % k == 0) {
            int ok = 1;
            for(int i = 0; i < n; i++)
                if(s[i] != s[i % k]) { ok = 0; break; }
            if(ok) { printf("%d\n", k); break; }
        }
    }
    return 0;
}
```
