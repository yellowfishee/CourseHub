// 题目1：判断回文
#include <stdio.h>
#include <string.h>
int main() {
    char s[100];
    scanf("%s", s);
    int len = strlen(s);
    int ok = 1;
    for(int i = 0; i < len/2; i++)
        if(s[i] != s[len-1-i]) { ok = 0; break; }
    printf(ok ? "YES\n" : "NO\n");
    return 0;
}

// 题目2：统计数字个数
#include <stdio.h>
int main() {
    char s[100];
    scanf("%s", s);
    int cnt = 0;
    for(int i = 0; s[i]; i++)
        if(s[i] >= '0' && s[i] <= '9') cnt++;
    printf("%d\n", cnt);
    return 0;
}

// 题目3：大写转小写
#include <stdio.h>
#include <ctype.h>
int main() {
    char s[100];
    scanf("%s", s);
    for(int i = 0; s[i]; i++)
        s[i] = tolower(s[i]);
    printf("%s\n", s);
    return 0;
}

// 习题4-1 鸡兔同笼
#include <stdio.h>
int main() {
    int n;
    scanf("%d", &n);
    int rabbit = n / 4;
    int chicken = (n - rabbit * 4) / 2;
    printf("%d %d\n", chicken, rabbit);
    return 0;
}

// 习题4-2 打鱼还是晒网
#include <stdio.h>
int main() {
    int y, m, d;
    scanf("%d %d %d", &y, &m, &d);
    int days = 0;
    for(int i = 1900; i < y; i++) {
        if((i % 4 == 0 && i % 100 != 0) || i % 400 == 0)
            days += 366;
        else
            days += 365;
    }
    int mdays[] = {0,31,28,31,30,31,30,31,31,30,31,30};
    for(int i = 1; i < m; i++) days += mdays[i];
    if((y % 4 == 0 && y % 100 != 0) || y % 400 == 0)
        if(m > 2) days += 1;
    days += d;
    int mod = days % 5;
    if(mod >= 1 && mod <= 3) printf("打鱼\n");
    else printf("晒网\n");
    return 0;
}
