## 参考答案

**填空题：**

**编程题（参考答案）：**

```cpp
// 题目1
#include <stdio.h>
int main() {
    int n, bai, shi, ge;
    scanf("%d", &n);
    bai = n / 100;
    shi = n / 10 % 10;
    ge = n % 10;
    printf("百位：%d\n", bai);
    printf("十位：%d\n", shi);
    printf("个位：%d\n", ge);
    return 0;
}

// 题目2
#include <stdio.h>
int main() {
    int n, bai, shi, ge;
    scanf("%d", &n);
    bai = n / 100;
    shi = n / 10 % 10;
    ge = n % 10;
    printf("%d%d%d\n", ge, shi, bai);
    return 0;
}

// 题目3
#include <stdio.h>
int main() {
    int a = 10, b = 20, t;
    t = a;
    a = b;
    b = t;
    printf("a = %d\n", a);
    printf("b = %d\n", b);
    return 0;
}

// 习题1-2 温度
#include <stdio.h>
int main() {
    double f, c;
    scanf("%lf", &f);
    c = 5.0 * (f - 32.0) / 9.0;
    printf("%.3f\n", c);
    return 0;
}

// 习题1-3 连续和
#include <stdio.h>
int main() {
    int n, sum = 0;
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) {
        sum += i;
    }
    printf("%d\n", sum);
    return 0;
}

// 习题1-5 打折
#include <stdio.h>
int main() {
    int n;
    scanf("%d", &n);
    double total = n * 95.0;
    if (total >= 300.0) {
        total *= 0.85;
    }
    printf("%.2f\n", total);
    return 0;
}

// 习题1-4 正弦和余弦
#include <stdio.h>
#include <math.h>
int main() {
    int n;
    scanf("%d", &n);
    double rad = n * acos(-1.0) / 180.0;
    printf("%.2f %.2f\n", sin(rad), cos(rad));
    return 0;
}
```
